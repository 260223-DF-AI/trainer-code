import torch
import torch.nn as nn

class SimpleLSTMModel(nn.Module):
    def __init__(self, vocab_size, embedding_dim, hidden_dim, output_classes):
        super().__init__()
        # 1. Embedding Layer: Converts integers to dense vectors
        self.embedding = nn.Embedding(num_embeddings=vocab_size, embedding_dim=embedding_dim)
        
        # 2. LSTM Layer: Processes the sequence
        self.lstm = nn.LSTM(input_size=embedding_dim, hidden_size=hidden_dim, batch_first=True)
        
        # 3. Final classification head
        self.fc = nn.Linear(hidden_dim, output_classes)
        
    def forward(self, x):
        embedded = self.embedding(x) 
        lstm_out, (h_n, c_n) = self.lstm(embedded)
        
        # We only care about the hidden state from the very last sequence step
        final_memory = h_n.squeeze(0) 
        
        return self.fc(final_memory)

# Assuming a vocab of 1000 words
model = SimpleLSTMModel(vocab_size=1000, embedding_dim=64, hidden_dim=128, output_classes=2)
dummy_text_batch = torch.randint(0, 1000, (32, 50)) # Batch of 32 sentences, 50 words each
predictions = model(dummy_text_batch)
